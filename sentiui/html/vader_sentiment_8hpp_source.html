167inline bool isupper(String word)
168{
169 return std::all_of(word.begin(), word.end(), [](unsigned char c) { return (!::isalpha(c)) || ::isupper(c); }) && std::any_of(word.begin(), word.end(), [](unsigned char c) { return ::isalpha(c); });

258 static double normalize(double score, double alpha=15)
259 {
260 // Normalize the score to be between -1 and 1 using an alpha that approximates the max expected value
261 double norm_score = score / sqrt((score * score) + alpha); // TODO: Look into fastest implementations of ISR
262 if (norm_score < -1.0)
263 return -1;
264 else if (norm_score > 1.0)
265 return 1;
266 return norm_score;
267 }

269 static bool allcap_differential(std::vector<String> words)
270 {
271 /*
272 Check whether just some words in the input are ALL CAPS
273 :param vector words: The words to inspect
274 :returns: `true` if some but not all items in `words` are ALL CAPS
275 */
276 int allcap_words = 0;
277 for (String word : words)
278 if (isupper(word))
279 allcap_words++;
280 int cap_differential = words.size() - allcap_words; // this is actually possibly buggy, behavior on emojis/emoticons are unknown TODO: investigate
281 return 0 < cap_differential && cap_differential < words.size(); // more space efficient than storing in a variable
282 }

284 static double scalar_inc_dec(String word, double valence, bool is_cap_diff)
285 {
286 // Check if the preceding words increase, decrease, or negate/nullify the valence
287 double scalar = 0.0;
288 String oword = word;
289 std::transform(word.begin(), word.end(), word.begin(), ::tolower);
290 if (BOOSTER_DICT.count(word))
291 {
292 scalar = BOOSTER_DICT[word];
293 if (valence < 0)
294 scalar *= -1;
295 // check if booster/dampener word is ALLCAPS (while others aren't)
296 if (is_cap_diff && isupper(oword))
297 {
298 if (valence > 0) // should be scalar > 0?
299 scalar += C_INCR;
300 else
301 scalar -= C_INCR;
302 }
303 }
304 return scalar;
